 68    def eventFilter(self, source, event):
 69        if event.type() == QEvent.KeyPress and event.key() in (Qt.Key_Return, Qt.Key_Enter):
 70            if source == self.ventana.input_buscar_nombre:
 71                texto = source.text().strip()
 72                cliente = self.dict_nombres.get(texto)
 73                if cliente:
 74                    self.rellenar_campos(cliente)
 75                    return True
 76            elif source == self.ventana.input_buscar_dni:
 77                texto = source.text().strip().upper()
 78                cliente = self.dict_dnis.get(texto)
 79                if cliente:
 80                    self.rellenar_campos(cliente)
 81                    return True
 82            elif source == self.ventana.input_buscar_telefono:
 83                texto = source.text().strip()
 84                cliente = self.dict_telefonos.get(texto)
 85                if cliente:
 86                    self.rellenar_campos(cliente)
 87                    return True
 88        return super().eventFilter(source, event)
