controladores.vehiculos_controlador

Controlador para la gestión de vehículos en el sistema.

Este controlador conecta la vista VentanaVehiculos con los modelos de datos, permitiendo:

  • Buscar, crear, modificar y eliminar vehículos.
  • Buscar clientes por nombre o DNI.
  • Crear nuevos clientes desde un diálogo si no existen.
  • Autocompletar campos con datos existentes.
  • Validar formularios antes de realizar operaciones.
  • Gestionar interacciones entre vehículos y clientes.

119    def eventFilter(self, source, event):
120        """
121        Captura eventos de teclado para autocompletar clientes al pulsar Enter.
122
123        Args:
124            source: Campo de texto que origina el evento.
125            event (QEvent): Evento recibido.
126
127        Returns:
128            bool: True si el evento fue manejado, False en caso contrario.
129        """
130        if event.type() == QEvent.KeyPress:
131            if event.key() in (Qt.Key_Return, Qt.Key_Enter):
132                if source == self.ventana.input_buscar_nombre:
133                    self.buscar_cliente_por_nombre()
134                    return True
135                elif source == self.ventana.input_buscar_dni:
136                    self.buscar_cliente_por_dni()
137                    return True
138        return super().eventFilter(source, event)

256        """
257        Busca un vehículo por matrícula y muestra sus datos junto con los del cliente asociado.

259        Args:
260            matricula (str): Matrícula del vehículo.
261        """

271        # Cliente
272        cliente = obtener_cliente_por_id(datos["cliente_id"])
273        if cliente:
274            nombre_completo = f"{cliente['nombre']} {cliente['primer_apellido']} {cliente.get('segundo_apellido', '')}".strip(
275            )

279        # Vehículo
280        self.ventana.input_matricula.setText(datos["matricula"])

282        self.ventana.input_modelo.setText(datos["modelo"])
283        self.ventana.input_color.setText(datos["color"])
284        self.ventana.input_anyo.setText(str(datos["anyo"]))
285        self.ventana.input_observaciones.setPlainText(
286            datos["observaciones"] or "")
287
288        self.ventana.input_numero_bastidor.setText(
289            datos["numero_bastidor"] or "")
290
291        # Seleccionar combustible
292        index_combustible = self.ventana.combo_combustible.findData(
293            datos["combustible_id"])

295            self.ventana.combo_combustible.setCurrentIndex(index_combustible)

299        # Seleccionar categoría

302        )
303        if indice_categoria >= 0:

305
306            # ⬇️ Actualiza combo de tipo antes de seleccionar
307            self.actualizar_tipos_por_categoria()
308
309            # 🟦 Seleccionar tipo
310            indice_tipo = self.ventana.combo_tipo.findText(
311                datos["tipo_nombre"], Qt.MatchFixedString
312            )
313            if indice_tipo >= 0:
314                self.ventana.combo_tipo.setCurrentIndex(indice_tipo)
315
316        self.ventana.boton_modificar.setEnabled(True)
317        self.ventana.boton_eliminar.setEnabled(True)

394    def obtener_datos_y_validar(self):
395        """
396        Extrae y valida los datos introducidos en el formulario.
397
398        Returns:
399            bool: True si todos los campos obligatorios son válidos, False en caso contrario.
400        """
401        nombre_cliente = self.ventana.input_buscar_nombre.text().strip().upper()
402        self.matricula = self.ventana.input_matricula.text().strip().upper()
403        self.marca = self.ventana.input_marca.text().strip()
404        self.modelo = self.ventana.input_modelo.text().strip()
405        self.color = self.ventana.input_color.text().strip()
406        self.anio = self.ventana.input_anyo.text().strip()
407        self.tipo = self.ventana.combo_tipo.currentIndex()
408        self.numero_bastidor = self.ventana.input_numero_bastidor.text().strip().upper()
409        self.combustible_id = self.ventana.combo_combustible.currentData()
410        self.observaciones = self.ventana.input_observaciones.toPlainText().strip()
411
412        if not nombre_cliente or nombre_cliente not in self.clientes_dict_nombre:
413            self.mostrar_error("Debes seleccionar un cliente válido.")
414            return False
415
416        self.cliente_id_actual = self.clientes_dict_nombre[nombre_cliente]
417
418        if not self.matricula or not self.marca or not self.modelo:
419            self.mostrar_error("Matrícula, marca y modelo son obligatorios.")
420            return False
421
422        if not self.numero_bastidor:
423            self.mostrar_error("El número de bastidor es obligatorio.")
424            return False
425
426        if self.combustible_id == -1:
427            self.mostrar_error("Debes seleccionar un tipo de combustible.")
428            return False
429
430        if self.tipo <= 0:
431            self.mostrar_error("Debes seleccionar un tipo de vehículo.")
432            return False
433
434        return True

121        """
