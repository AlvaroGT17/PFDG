207    def actualizar_tabla_vehiculos(self):
208        def coincide_rango(valor, texto_rango):
209            try:
210                val = float(valor)
211                if texto_rango.endswith("+"):
212                    return val >= float(texto_rango[:-1].replace("k", "000"))
213                minimo, maximo = texto_rango.replace("k", "000").split("-")
214                return float(minimo) <= val < float(maximo)
215            except:
216                return False
217
218        self.vista.vehiculos_filtrados = []
219
220        for veh in self.vista.vehiculos_disponibles:
221            if veh.get("estado") not in ("DISPONIBLE", "RESERVADO"):
222                continue
223
224            incluir = True
225            for campo, combo in self.vista.filtros.items():
226                filtro = combo.currentText()
227                if filtro == "Cualquiera":
228                    continue
229                valor = str(veh.get(campo, ""))
230
231                if campo in ("kilometros", "potencia_cv", "precio_venta"):
232                    if not coincide_rango(valor, filtro):
233                        incluir = False
234                        break
235                else:
236                    if filtro != valor:
237                        incluir = False
238                        break
239
240            if incluir:
241                self.vista.vehiculos_filtrados.append(veh)
242
243        self.cargar_tabla_filtrada()

231
232                if campo in ("kilometros", "potencia_cv", "precio_venta"):
233                    if not coincide_rango(valor, filtro):
234                        incluir = False
