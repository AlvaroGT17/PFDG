213    def recepcion_seleccionada(self, index):
214        """
215        Actualiza los campos visibles según la recepción seleccionada en el combo box.
216
217        Parámetros:
218            index (int): Índice del combo box que indica qué recepción fue seleccionada.
219        """
220        if index <= 0 or not hasattr(self, "controlador"):
221            self.campo_cliente.clear()

225            self.texto_observaciones.clear()

227            return
228
229        datos = self.controlador.recepciones[index - 1]
230        self.campo_cliente.setText(datos.get("cliente", ""))
231        self.campo_vehiculo.setText(datos.get("matricula", ""))
232        fecha = datos.get("fecha", None)
233        self.campo_fecha.setText(
234            fecha.strftime("%d/%m/%Y %H:%M") if isinstance(fecha,
235                                                           datetime) else "(sin fecha)"
236        )
237
238        precio_max = datos.get("precio_max_autorizado", None)
239        self.campo_limite.setText(f"{precio_max:.2f} €" if precio_max else "")
240        self.texto_observaciones.setText(datos.get("observaciones", ""))
241
242        self._actualizar_estado_guardar()

266    def eliminar_tarea_seleccionada(self):

277    def actualizar_coste_total(self):
278        """
279        Suma los valores de la columna "Total" de la tabla y actualiza el campo de coste total.
280
281        También compara el total con el límite autorizado y actualiza la respuesta sugerida del cliente.
282        """
283        total = 0
284        for fila in range(self.tabla_tareas.rowCount()):
285            item = self.tabla_tareas.item(fila, 3)
286            if item:
287                texto = item.text().replace("€", "").strip()
288                try:
289                    total += float(texto)
290                except ValueError:
291                    pass
292        self.campo_coste_total.setText(f"{total:.2f} €")
293
294        try:
295            limite = float(self.campo_limite.text().replace("€", "").strip())
296            if total <= limite:
297                self.combo_respuesta.setCurrentText("Aceptado")
298            else:
299                self.combo_respuesta.setCurrentText("En espera")
300        except ValueError:
301            self.combo_respuesta.setCurrentText("")

303    def subir_fila(self):
304        """
305        Mueve hacia arriba la tarea actualmente seleccionada en la tabla, si no está en la primera fila.
306        """
307        fila = self.tabla_tareas.currentRow()
308        if fila > 0:
309            self._intercambiar_filas(fila, fila - 1)
310            self.tabla_tareas.selectRow(fila - 1)

312    def bajar_fila(self):
313        """
314        Mueve hacia abajo la tarea actualmente seleccionada en la tabla, si no está en la última fila.
315        """
316        fila = self.tabla_tareas.currentRow()
317        if fila != -1 and fila < self.tabla_tareas.rowCount() - 1:
318            self._intercambiar_filas(fila, fila + 1)
319            self.tabla_tareas.selectRow(fila + 1)

321    def _intercambiar_filas(self, fila1, fila2):
322        """
323        Intercambia los valores entre dos filas de la tabla de tareas.
324
325        Parámetros:
326            fila1 (int): Índice de la primera fila.
327            fila2 (int): Índice de la segunda fila.
328        """
329        for col in range(self.tabla_tareas.columnCount()):
330            texto1 = self.tabla_tareas.item(fila1, col).text()
331            texto2 = self.tabla_tareas.item(fila2, col).text()
332            self.tabla_tareas.setItem(fila1, col, QTableWidgetItem(texto2))
333            self.tabla_tareas.setItem(fila2, col, QTableWidgetItem(texto1))

335    def _actualizar_estado_guardar(self):
336        """
337        Activa o desactiva el botón de guardar presupuesto según si hay una recepción válida
338        y al menos una tarea cargada en la tabla.
339        """
340        index = self.combo_recepciones.currentIndex()
341        tiene_recepcion_valida = index > 0 and hasattr(self, "controlador")
342        tiene_tareas = self.tabla_tareas.rowCount() > 0
343        self.boton_guardar.setEnabled(tiene_recepcion_valida and tiene_tareas)
344        self.boton_anadir_tarea.setEnabled(tiene_recepcion_valida)

346    def guardar_presupuesto(self):
347        """
348        Simula el guardado del presupuesto, mostrando en consola las opciones seleccionadas

379        self.texto_observaciones.setText(
380            datos_presupuesto.get("observaciones", ""))
381
382        self.tabla_tareas.setRowCount(0)

392            self.tabla_tareas.setItem(
393                fila, 3, QTableWidgetItem(f"{tarea['total']:.2f} €"))

278    def actualizar_coste_total(self):
279        """
280        Suma los valores de la columna "Total" de la tabla y actualiza el campo de coste total.
281
282        También compara el total con el límite autorizado y actualiza la respuesta sugerida del cliente.
283        """
284        total = 0
285        for fila in range(self.tabla_tareas.rowCount()):
286            item = self.tabla_tareas.item(fila, 3)
287            if item:
288                texto = item.text().replace("€", "").strip()
289                try:
290                    total += float(texto)
291                except ValueError:
292                    pass
293        self.campo_coste_total.setText(f"{total:.2f} €")
294
295        try:
296            limite = float(self.campo_limite.text().replace("€", "").strip())
297            if total <= limite:
298                self.combo_respuesta.setCurrentText("Aceptado")
299            else:
300                self.combo_respuesta.setCurrentText("En espera")
301        except ValueError:
302            self.combo_respuesta.setCurrentText("")

318        if fila != -1 and fila < self.tabla_tareas.rowCount() - 1:
319            self._intercambiar_filas(fila, fila + 1)

333            self.tabla_tareas.setItem(fila1, col, QTableWidgetItem(texto2))
334            self.tabla_tareas.setItem(fila2, col, QTableWidgetItem(texto1))
