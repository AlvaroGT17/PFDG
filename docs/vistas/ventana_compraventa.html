 22    def actualizar_secciones(self, tipo):
 23        """
 24        Activa y muestra únicamente las secciones correspondientes al tipo de operación
 25        seleccionado por el usuario: "compra" o "venta".
 26
 27        Si el valor recibido es el texto por defecto, se ocultan todas las secciones.
 28        Este método gestiona la visibilidad, habilitación y expansión de las secciones:
 29        - Cliente (común a ambas operaciones)
 30        - Vehículo (solo para compras)
 31        - Operación (solo para ventas)
 32
 33        Parámetros:
 34            tipo (str): Tipo de operación seleccionado desde el combo. Se espera
 35                        "Compra por parte del concesionario" o "Venta por parte del concesionario".
 36        """
 37        tipo = tipo.upper()
 38
 39        # Ocultar todo de inicio
 40        for seccion in [self.seccion_cliente, self.seccion_vehiculo, self.seccion_operacion]:
 41            seccion['grupo'].setEnabled(False)
 42            seccion['contenido'].setEnabled(False)
 43            seccion['contenido'].setVisible(False)
 44            seccion['toggle'].setChecked(False)
 45            seccion['toggle'].setEnabled(False)
 46
 47        if tipo == "SELECCIONE LA OPERACIÓN DESEADA":
 48            return
 49
 50        # Activar sección Cliente (común a ambas)
 51        self.seccion_cliente['grupo'].setEnabled(True)
 52        self.seccion_cliente['contenido'].setEnabled(True)
 53        self.seccion_cliente['toggle'].setEnabled(True)
 54        self.seccion_cliente['contenido'].setVisible(True)
 55        self.seccion_cliente['toggle'].setChecked(True)
 56
 57        if tipo == "COMPRA POR PARTE DEL CONCESIONARIO":
 58            self.seccion_vehiculo['grupo'].setEnabled(True)
 59            self.seccion_vehiculo['contenido'].setEnabled(True)
 60            self.seccion_vehiculo['toggle'].setEnabled(True)
 61            self.seccion_vehiculo['contenido'].setVisible(True)
 62            self.seccion_vehiculo['toggle'].setChecked(True)
 63
 64        elif tipo == "VENTA POR PARTE DEL CONCESIONARIO":
 65            self.seccion_operacion['grupo'].setEnabled(True)
 66            self.seccion_operacion['contenido'].setEnabled(True)
 67            self.seccion_operacion['toggle'].setEnabled(True)
 68            self.seccion_operacion['contenido'].setVisible(True)
 69            self.seccion_operacion['toggle'].setChecked(True)

 98    def crear_seccion_plegable(self, titulo):
 99        """
100        Crea una sección visual plegable (expandible/colapsable) con un título especificado.
101
102        Esta sección consiste en un `QGroupBox` que contiene un botón de tipo `QToolButton` 
103        que alterna entre expandir y contraer el contenido. Se utiliza para organizar
104        visualmente partes de la interfaz como los datos del cliente, del vehículo, etc.
105
106        Parámetros:
107            titulo (str): Título que se mostrará en la cabecera de la sección.
108
109        Retorna:
110            dict: Un diccionario con claves:
111                - 'grupo': el QGroupBox principal de la sección.
112                - 'contenido': el QWidget que contiene los campos plegables.
113                - 'toggle': el botón para expandir/colapsar la sección.
114        """
115        grupo = QGroupBox()
116        layout = QVBoxLayout(grupo)
117
118        cabecera = QHBoxLayout()
119        boton_toggle = QToolButton()
120        icono_expandir = QIcon(obtener_ruta_absoluta("img/mas.png"))
121        icono_colapsar = QIcon(obtener_ruta_absoluta("img/menos.png"))
122
123        boton_toggle.setIcon(icono_expandir)
124        boton_toggle.setCheckable(True)
125        boton_toggle.setChecked(False)
126        boton_toggle.setIconSize(QSize(24, 24))
127
128        etiqueta = QLabel(f"<b>{titulo}</b>")
129        etiqueta.setSizePolicy(QSizePolicy.Expanding, QSizePolicy.Preferred)
130        etiqueta.setStyleSheet("color: white")
131
132        cabecera.addWidget(etiqueta)
133        cabecera.addWidget(boton_toggle)
134
135        contenido = QWidget()
136        contenido.setObjectName("contenido_plegable")
137        contenido.setVisible(False)
138
139        # El toggle controla visibilidad del contenido
140        def toggle():
141            visible = boton_toggle.isChecked()
142            contenido.setVisible(visible)
143            boton_toggle.setIcon(icono_colapsar if visible else icono_expandir)
144
145        boton_toggle.clicked.connect(toggle)
146
147        layout.addLayout(cabecera)
148        layout.addWidget(contenido)
149
150        return {
151            "grupo": grupo,
152            "contenido": contenido,
153            "toggle": boton_toggle
154        }

763    def eventFilter(self, source, event):
764        """
765        Activa o desactiva el modo de firma en el capturador correspondiente (compra o venta).
766
767        Este método se utiliza para permitir que el cliente firme en el cuadro de firma, ya sea
768        para la operación de compra o de venta. Cambia el estado visual, activa o desactiva la
769        funcionalidad del capturador, muestra u oculta el mensaje de firma y modifica el texto
770        del botón que controla la acción.
771
772        Parámetros:
773            capturador (CapturadorFirma): Widget de firma a activar o desactivar.
774            boton (QToolButton): Botón que ha activado el proceso y cuyo texto se modifica según el estado.
775        """
776        if event.type() == QEvent.KeyPress and event.key() in (Qt.Key_Enter, Qt.Key_Return):
777            if self.firma_activa_compra:
778                self.toggle_firma(self.capturador_firma,
779                                  self.boton_activar_firma)
780                return True
781            elif self.firma_activa_venta:
782                self.toggle_firma(self.capturador_firma_venta,
783                                  self.boton_activar_firma_venta)
784                return True
785        return super().eventFilter(source, event)
