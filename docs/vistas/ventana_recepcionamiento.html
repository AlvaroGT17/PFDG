154    def crear_seccion_plegable(self, titulo):
155        """
156        Crea una sección plegable con encabezado e ícono de expansión/colapso.
157
158        :param titulo: Título de la sección.
159        :return: Diccionario con los elementos 'grupo', 'contenido' y 'toggle'.
160        """
161        grupo = QGroupBox()
162        layout = QVBoxLayout(grupo)
163
164        cabecera = QHBoxLayout()
165        boton_toggle = QToolButton()
166        icono_expandir = QIcon(obtener_ruta_absoluta("img/mas.png"))
167        icono_colapsar = QIcon(obtener_ruta_absoluta("img/menos.png"))
168        boton_toggle.setIcon(icono_expandir)
169        boton_toggle.setCheckable(True)
170        boton_toggle.setChecked(False)
171        boton_toggle.setIconSize(QSize(24, 24))
172        etiqueta = QLabel(f"<b>{titulo}</b>")
173        etiqueta.setSizePolicy(QSizePolicy.Expanding, QSizePolicy.Preferred)
174        etiqueta.setStyleSheet("color: white")
175
176        cabecera.addWidget(etiqueta)
177        cabecera.addWidget(boton_toggle)
178
179        contenido = QWidget()
180        contenido.setObjectName("contenido_plegable")
181        contenido.setVisible(False)
182
183        def toggle():
184            """
185            Alterna la visibilidad del contenido de la sección plegable.
186
187            Cambia el estado visible/invisible del contenido del grupo y ajusta el ícono
188            del botón de expansión dependiendo de si está desplegado o no.
189            """
190            visible = boton_toggle.isChecked()
191            contenido.setVisible(visible)
192            boton_toggle.setIcon(icono_colapsar if visible else icono_expandir)
193
194        boton_toggle.clicked.connect(toggle)
195
196        layout.addLayout(cabecera)
197        layout.addWidget(contenido)
198
199        return {"grupo": grupo, "contenido": contenido, "toggle": boton_toggle}
